#简单的word count程序
```
#!/usr/bin/env python
#-*- coding:utf8 -*-

from sys import stdin

data = stdin.read()
chars = len( data )
words = len( data.split() )
lines = data.count('\n')

"""locals()在这里返回一个字典的数据类型，
前面的%(lines)s，lines表示是后面字典的key对应的值以字符串的形式取出
如果在后面的字典中找不到对应的key，那么则会报个KeyErr错误"""
print "%(lines)s%(words)s%(chars)s" % locals()
print "%(lines)s%(words)s%(chars)s" % {"lines":lines, "words":words, "chars":chars}
```
