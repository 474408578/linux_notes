#多重继承
- python支持多重继承，即一个类可以继承多个父类
- 语法
class class_name(Parent_c1, Parent_c2, ...)
- `注意`
当父类中出现多个自定义的__init__方法时，多重继承只执行`第一个类`的__init__方法，其它的不执行

```
5 #!/usr/bin/env python
4 #-*- coding:utf-8 -*-
3 #时间：2016-10-16 Sun 21:46
2 #运行结果：
1 """
0 ============================================================
1 ************************************************************
2 This is a public method. ----->People.think()
3 HongKong AAAAAAAAAAAAAA
4 """
5 
6 class Person(object):
7     color = 'grown'
8 
9     def __init__(self):
10         self.dwell = 'HongKong'
11         self.color = 'AAAAAAAAAAAAAA'
12 
13     def think(self):
14         print 'This is a public method. ----->Person.think()'
15 
16 class People(object):
17     sea = 'Water'
18 
19     def __init__(self):
20         self.dwell = 'East sea'
21 
22     def think(self):
23         print 'This is a public method. ----->People.think()'
24 
25 #class Chinese(Person, People):
26 class Chinese(People, Person):
27     def __init__(self):
28         Person.__init__(self)
29 
30 print '===' * 20
31 changFeng = Chinese()
32 print '***' * 20
33 changFeng.think()
34 print changFeng.dwell, changFeng.color
```
